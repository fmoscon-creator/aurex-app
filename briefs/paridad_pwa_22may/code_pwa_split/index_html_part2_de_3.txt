function pacSaveNombre(){
  var sb=window._supabase;
  var ni=document.getElementById('input-nombre');

  var nombre=ni.value.trim();

  });

  var ct=document.getElementById('cel-txt-b6');
  var msg=document.getElementById('auth-msg');
  var sb=window._supabase;
  if(!sb||!ci) return;
  var cel=ci.value.trim();
  if(!cel){
    if(msg){msg.textContent='Ingres&#225; un n&#250;mero';msg.style.color='#e74c3c';}
    return;
  }
  sb.auth.updateUser({data:{celular:cel}}).then(function(r){
    if(r.error){
      if(msg){msg.textContent='Error al guardar';msg.style.color='#e74c3c';}
    } else {
      if(window._currentUser&&window._currentUser.user_metadata)
        window._currentUser.user_metadata.celular=cel;
      var ci2=document.getElementById('celular-input');
      if(ci2) ci2.value=cel;
      var ct2=document.getElementById('auth-celular-txt');
      if(ct2) ct2.textContent=cel;
      if(ct) ct.textContent=cel;
      if(msg){msg.textContent='Celular guardado';msg.style.color='#2ecc71';}
      setTimeout(function(){if(msg)msg.textContent='';},3000);
    }
  });
}
function pacInitSeg(){
  var ua=navigator.userAgent;
  var dev='';
  if(/iPhone/.test(ua)) dev='iPhone';
  else if(/Android/.test(ua)) dev='Android';
  else if(/iPad/.test(ua)) dev='iPad';
  else dev=window.t ? t('navegador_web') : 'Navegador web';
  var el=document.getElementById('seg-dispositivo');
  var el2=document.getElementById('seg-session-info');
  if(el) el.textContent=dev;
  if(el2){
    var now=new Date();
    var hoy=window.t ? t('hoy') : 'Hoy';
    el2.textContent=hoy+' '+now.getHours()+':'+('0'+now.getMinutes()).slice(-2);
  }
  var acc=document.getElementById('seg-accesos');
  if(acc){
    var d=new Date();
    var fmt=function(dt){
      return dt.getDate()+'/'+(dt.getMonth()+1)+'/'+dt.getFullYear()+
        ' '+dt.getHours()+':'+('0'+dt.getMinutes()).slice(-2);
    };

      '<div style="display:flex;justify-content:space-between;font-size:12px;color:var(--textSec)">'+
      '<span>'+fmt(d)+'</span><span style="color:var(--textDim)">'+dev+'</span></div>'+
      '<div style="display:flex;justify-content:space-between;font-size:12px;color:var(--textDim)">'+
      '<span>'+fmt(d2)+'</span><span style="color:var(--border2)">'+dev+'</span></div>'+
      '<div style="display:flex;justify-content:space-between;font-size:12px;color:var(--border2)">'+
      '<span>'+fmt(d3)+'</span><span style="color:var(--border)">'+dev+'</span></div>';
  }

function initPerfilCanvas(){
  var canvas=document.getElementById('perfil-canvas');
  if(!canvas) return;
  var screen=document.getElementById('screen-perfil');
  canvas.width=screen.offsetWidth||390;
  canvas.height=screen.offsetHeight||700;
  var ctx=canvas.getContext('2d');
  var W=canvas.width, H=canvas.height;
  var stars=[];
  for(var i=0;i<70;i++){
    stars.push({
      x:Math.random()*W,
      y:Math.random()*H,
      r:Math.random()*1.4+0.3,
      alpha:Math.random(),
      da:(Math.random()*0.008+0.003)*(Math.random()<0.5?1:-1),
      vx:(Math.random()-0.5)*0.08,
      vy:(Math.random()-0.5)*0.08
    });
  }
  function draw(){
    ctx.clearRect(0,0,W,H);
    // Lineas entre estrellas cercanas
    for(var i=0;i<stars.length;i++){
      for(var j=i+1;j<stars.length;j++){
        var dx=stars[i].x-stars[j].x;
        var dy=stars[i].y-stars[j].y;
        var dist=Math.sqrt(dx*dx+dy*dy);
        if(dist<90){
          var lineAlpha=(1-dist/90)*0.18;
          ctx.beginPath();
          ctx.moveTo(stars[i].x,stars[i].y);
          ctx.lineTo(stars[j].x,stars[j].y);
          ctx.strokeStyle='rgba(212,160,23,'+lineAlpha+')';
          ctx.lineWidth=0.5;
          ctx.stroke();
        }
      }
    }
    // Estrellas
    for(var i=0;i<stars.length;i++){
      var s=stars[i];
      ctx.beginPath();
      ctx.arc(s.x,s.y,s.r,0,Math.PI*2);
      ctx.fillStyle='rgba(212,160,23,'+Math.max(0,Math.min(1,s.alpha))+')';
      ctx.fill();
      // Actualizar
      s.alpha+=s.da;
      if(s.alpha>1||s.alpha<0) s.da*=-1;
      s.x+=s.vx; s.y+=s.vy;
      if(s.x<0) s.x=W; if(s.x>W) s.x=0;
      if(s.y<0) s.y=H; if(s.y>H) s.y=0;
    }
    requestAnimationFrame(draw);
  }
  draw();
}
